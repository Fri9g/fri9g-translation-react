## Install

```bash
npm install --save fri9g-translation-react
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'fri9g-translation-react'
import 'fri9g-translation-react/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
